55 uint8_t Ipl_ParseIpf(Ipl_IpfData_t *ipf, uint32_t lData, uint8_t pData[], uint8_t stringType)
56 {
57  uint32_t offset = 6U;
58  uint8_t res = IPL_RES_ERR_IPF_INVALID;
59  uint32_t offset_s;
60  uint8_t ptype;
61  uint32_t i, pi, pid, pval, plen;
62  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf called with %u byte IPF, StringType 0x%02X", lData, stringType);
63  res = Ipl_ClrPData(lData, pData);
64  if (IPL_RES_OK == res)
65  {
66  if ((STRING_MIN_LEN <= lData) && (NULL != pData))
67  {
68  ipf->ChipID = Ipl_PData(1U, lData, pData);
69  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf IPF ChipID 0x%02X", ipf->ChipID);
70  res = IPL_RES_ERR_IPF_WRONGINIC;
71  if (Ipl_IplData.ChipID == ipf->ChipID)
72  {
73  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf start with Offset %u", offset);
74  while ((offset + 9U) < lData)
75  {
76  if (Ipl_PData(offset, lData, pData) == stringType)
77  {
78  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf StringType 0x%02X found at Offset %u", stringType, offset);
79  break;
80  }
81  offset_s = offset;
82  offset += ( ((uint32_t) Ipl_PData(offset_s + 6U, lData, pData) << 24U )
83  + ((uint32_t) Ipl_PData(offset_s + 7U, lData, pData) << 16U )
84  + ((uint32_t) Ipl_PData(offset_s + 8U, lData, pData) << 8U )
85  + ((uint32_t) Ipl_PData(offset_s + 9U, lData, pData) & 0xFFU) ) + 10U;
86  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf current Offset %u", offset);
87  }
88  res = IPL_RES_ERR_IPF_WRONGSTRINGTYPE; /* Offset out of bounds */
89  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf current Offset %u", offset);
90  if ((offset + 13U) < lData) /* +9U, 10U */
91  {
92  res = IPL_RES_OK;
93  if (Ipl_PData(offset, lData, pData) == stringType)
94  {
95  /* Image is fine and we can continue */
96  Ipl_IpfData.StringType = stringType;
97  Ipl_IpfData.StringOffset = offset + 10U; /* Points to first data byte */
98  Ipl_IpfData.StringSize = (uint32_t) Ipl_PData(offset + 6U, lData, pData) << 24U;
99  Ipl_IpfData.StringSize += (uint32_t) Ipl_PData(offset + 7U, lData, pData) << 16U;
100  Ipl_IpfData.StringSize += (uint32_t) Ipl_PData(offset + 8U, lData, pData) << 8U;
101  Ipl_IpfData.StringSize += (uint32_t) Ipl_PData(offset + 9U, lData, pData) ;
102  switch (ipf->StringType)
103  {
104  case STRINGTYPE_FW:
105  ipf->ProgAddr = (uint32_t) Ipl_PData(offset + 2U, lData, pData) << 24U ;
106  ipf->ProgAddr += (uint32_t) Ipl_PData(offset + 3U, lData, pData) << 16U ;
107  ipf->ProgAddr += (uint32_t) Ipl_PData(offset + 4U, lData, pData) << 8U ;
108  ipf->ProgAddr += (uint32_t) Ipl_PData(offset + 5U, lData, pData) ;
109  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf ProgAddr 0x%X", ipf->ProgAddr);
110  break;
111  case STRINGTYPE_PS:
112  ipf->ProgAddr = Ipl_IpfData.Meta.PatchsStdStartAddr;
113  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf ProgAddr 0x%X", ipf->ProgAddr);
114  break;
115  case STRINGTYPE_META:
116  ipf->Meta.NumOfItems = (uint32_t) Ipl_PData(offset + 10U, lData, pData) << 24U ;
117  ipf->Meta.NumOfItems += (uint32_t) Ipl_PData(offset + 11U, lData, pData) << 16U ;
118  ipf->Meta.NumOfItems += (uint32_t) Ipl_PData(offset + 12U, lData, pData) << 8U ;
119  ipf->Meta.NumOfItems += (uint32_t) Ipl_PData(offset + 13U, lData, pData) ;
120  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf %u Meta Items found", ipf->Meta.NumOfItems);
121  if ((offset + 14U + (ipf->Meta.NumOfItems*12U)) < lData)
122  {
123  for (i=0U; i<ipf->Meta.NumOfItems; i++)
124  {
125  pid = (uint32_t) Ipl_PData(offset + 14U + (i*12U) , lData, pData) << 24U;
126  pid += (uint32_t) Ipl_PData(offset + 14U + (i*12U) + 1U, lData, pData) << 16U;
127  pid += (uint32_t) Ipl_PData(offset + 14U + (i*12U) + 2U, lData, pData) << 8U;
128  pid += (uint32_t) Ipl_PData(offset + 14U + (i*12U) + 3U, lData, pData) ;
129  ptype = Ipl_PData(offset + 14U + (i*12U) + 4U, lData, pData) ;
130  pval = (uint32_t) Ipl_PData(offset + 14U + (i*12U) + 8U, lData, pData) << 24U;
131  pval += (uint32_t) Ipl_PData(offset + 14U + (i*12U) + 9U, lData, pData) << 16U;
132  pval += (uint32_t) Ipl_PData(offset + 14U + (i*12U) + 10U, lData, pData) << 8U;
133  pval += (uint32_t) Ipl_PData(offset + 14U + (i*12U) + 11U, lData, pData) ;
134 
135  res = Ipl_SetStdMetaProps(ipf, pid, pval, ptype);
136  /* Handle Tool Type String */
137  if (METAID_TOOLTYPE == pid)
138  {
139  if (METATYPE_STRING == ptype)
140  {
141  plen = (uint32_t) Ipl_PData(offset + 14U + (i*12U) + 5U, lData, pData) << 16U ;
142  plen += (uint32_t) Ipl_PData(offset + 14U + (i*12U) + 6U, lData, pData) << 8U ;
143  plen += (uint32_t) Ipl_PData(offset + 14U + (i*12U) + 7U, lData, pData) ;
144  if ((TOOL_MAX_TYPELEN - 1U) < plen)
145  {
146  plen = TOOL_MAX_TYPELEN - 1U;
147  }
148  if ((offset + 10U + pval + plen) <= lData)
149  {
150  for (pi=0U; pi<plen; pi++)
151  {
152  ipf->Meta.ToolType[pi] = (char) Ipl_PData(offset + 10U + pval + pi,
153  lData, pData);
154  }
155  ipf->Meta.ToolType[pi + 1U] = '\0';
156  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf MetaID 0x%08X (ToolType) = '%s'",
157  pid, ipf->Meta.ToolType);
158  }
159  else
160  {
161  /* res = IPL_RES_ERR_IPF_WRONGSTRINGTYPE; */ /* No error, just ignore it */
162  Ipl_Trace(IPL_TRACETAG_ERR,
163  "Ipl_ParseIpf MetaID 0x%08X (ToolType) out of bounds", pid);
164  }
165  }
166  }
167  }
168  }
169  else
170  {
171  res = IPL_RES_ERR_IPF_WRONGSTRINGTYPE;
172  Ipl_Trace(IPL_TRACETAG_ERR, "Ipl_ParseIpf Meta data out of bounds");
173  }
174  break;
175  case STRINGTYPE_CONFIG:
176  case STRINGTYPE_CS:
177  case STRINGTYPE_IS:
178  break;
179  default:
180  Ipl_Trace(IPL_TRACETAG_ERR, "Ipl_ParseIpf StringType 0x%02X unknown", ipf->StringType);
181  break;
182  }
183  if (IPL_RES_OK == res)
184  {
185  if (ipf->StringSize < lData)
186  {
187  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_ParseIpf StringSize %u", ipf->StringSize);
188  Ipl_TraceIpf(ipf->StringSize, pData);
189  }
190  else
191  {
192  res = IPL_RES_ERR_IPF_WRONGSTRINGTYPE;
193  Ipl_Trace(IPL_TRACETAG_ERR, "Ipl_ParseIpf StringSize out of bounds");
194  }
195  }
196  else
197  {
198  Ipl_Trace(IPL_TRACETAG_ERR, "Ipl_ParseIpf Internal size out of bounds");
199  }
200  }
201  }
202  else
203  {
204  res = IPL_RES_ERR_IPF_WRONGSTRINGTYPE;
205  Ipl_Trace(IPL_TRACETAG_ERR, "Ipl_ParseIpf StringType 0x%02X not found", stringType);
206  }
207  }
208  }
209  else
210  {
211  Ipl_Trace(IPL_TRACETAG_ERR, "Ipl_ParseIpf IPF contains no valid data");
212  }
213 #ifdef IPL_LEGACY_IPF
214  if (STRINGTYPE_META == stringType)
215  {
216  res = Ipl_SetDefaultMetaProps(ipf);
217  }
218 #endif
219  }
220  Ipl_Trace(Ipl_TraceTag(res), "Ipl_ParseIpf returned 0x%02X", res);
221  return res;
222 }

516 static uint8_t Ipl_SetDefaultMetaProps(Ipl_IpfData_t *ipf)

707 static void Ipl_TraceIpf(uint32_t nOfBytes, uint8_t pData[])
708 {
709 #ifdef IPL_TRACETAG_IPF
710  uint8_t i;
711  char line [TRACELINE_MAXLEN];
712  uint32_t len;
713  uint32_t data = 0U;
714  uint32_t maxBpl = (TRACELINE_MAXLEN - 4U) / 3U;
715  do
716  {
717  if (nOfBytes >= maxBpl)
718  {
719  len = maxBpl;
720  nOfBytes -= maxBpl;
721  }
722  else
723  {
724  len = nOfBytes;
725  nOfBytes = 0U;
726  }
727  /* Write Ipf line */
728  for (i=0U; i<len; i++)
729  {
730  line[0U+(i*3U)] = Ipl_BcdChr(Ipl_PData(Ipl_IpfData.StringOffset+i+data, nOfBytes, pData), IPL_HIGH);
731  line[1U+(i*3U)] = Ipl_BcdChr(Ipl_PData(Ipl_IpfData.StringOffset+i+data, nOfBytes, pData), IPL_LOW);
732  line[2U+(i*3U)] = ' ';
733  }
734  line[0U+(len*3U)] = '\0';
735  Ipl_Trace(IPL_TRACETAG_IPF, line);
736  if (0U == nOfBytes)
737  {
738  break;
739  }
740  data += maxBpl;
741  } while (nOfBytes != 0U);
742 #endif
743 }
