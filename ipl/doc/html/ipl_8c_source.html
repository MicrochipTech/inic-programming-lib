97 uint8_t Ipl_EnterProgMode(uint8_t chipID)
98 {
99  uint8_t res = IPL_RES_ERR_HW_INIC_COM;
100  uint8_t cc = 0U;
101  Ipl_IplData.ChipID = chipID;
102 #ifdef IPL_INICDRIVER_OPENCLOSE
103  cc = Ipl_InicDriverOpen();
104 #endif
105  Ipl_Trace(IPL_TRACETAG_INFO, "INIC Programming Library %s", VERSIONTAG);
106  Ipl_Trace(IPL_TRACETAG_INFO, "For support contact http://www.microchip.com/support");
107  Ipl_TraceCfg();
108  Ipl_Trace(IPL_TRACETAG_INFO, "Ipl_EnterProgMode called with ChipID 0x%02X", chipID);
109  Ipl_InicData.TestMemCleared = INIC_TESTMEM_UNCLEARED;
110  Ipl_ClrIpfData(&Ipl_IpfData);
111  if (0U == cc)
112  {
113  res = Ipl_StartupInic(INIC_MODE_BOOT);
114  if (IPL_RES_OK == res)
115  {
116  Ipl_ClrTel();
117  Ipl_IplData.Tel[0] = CMD_PROGSTART;
118  switch (chipID)
119  {
120  case IPL_CHIP_OS81118:
121  case IPL_CHIP_OS81119:
122  case IPL_CHIP_OS81050:
123  case IPL_CHIP_OS81060:
124  case IPL_CHIP_OS81082:
125  case IPL_CHIP_OS81092:
126  case IPL_CHIP_OS81110:
127  Ipl_IplData.TelLen = CMD_PROGSTART_TXLEN;
128  break;
129  case IPL_CHIP_OS81210:
130  case IPL_CHIP_OS81212:
131  case IPL_CHIP_OS81214:
132  case IPL_CHIP_OS81216:
133  Ipl_IplData.Tel[4] = 0x28U;
134  Ipl_IplData.Tel[5] = 0x1BU;
135  Ipl_IplData.Tel[6] = 0x6BU;
136  Ipl_IplData.Tel[7] = 0x95U;
137  Ipl_IplData.TelLen = 8U;
138  break;
139  default:
140  res = IPL_RES_ERR_NOT_SUPPORTED;
141  break;
142  }
143  if (IPL_RES_OK == res)
144  {
145  res = Ipl_ExecInicCmd();
146  if (IPL_RES_OK == res)
147  {
148  res = Ipl_ReadFirmwareVersion(); /* for later crosscheck of the IPF file */
149  res = IPL_RES_OK;
150  }
151  }
152  }
153  }
154  Ipl_ExportChipInfo();
155  Ipl_Trace(Ipl_TraceTag(res), "Ipl_EnterProgMode returned 0x%02X", res);
156  return res;
157 }

161 uint8_t Ipl_LeaveProgMode(void)
162 {
163  uint8_t res;
164  uint8_t cc;
165  res = Ipl_StartupInic(INIC_MODE_NORMAL);
166 #ifdef IPL_INICDRIVER_OPENCLOSE
167  if (IPL_RES_OK == res)
168  {
169  cc = Ipl_InicDriverClose();
170  if (0U != cc)
171  {
172  res = IPL_RES_ERR_HW_INIC_COM;
173  }
174  }
175 #endif
176  Ipl_Trace(Ipl_TraceTag(res), "Ipl_LeaveProgMode returned 0x%02X", res);
177  return res;
178 }
